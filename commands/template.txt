from discord import Embed
from discord.ext import commands
import errors
from data.users import get_user

info = {
    "name": "",
    "aliases": [""],
    "description": "",
    "parameters": "",
    "usages": [""],
    "import": False,
}

class CommandName(commands.Cog):
    def __init__(self, bot):
        self.bot = bot

    @commands.command(aliases=info["aliases"])
    async def commandname(self, ctx, *params):
        user = get_user(ctx)

        await run(ctx, user)

async def run(ctx, user):
    embed = Embed(
        title="Title",
        color=user["colors"]["embed"],
    )

    await ctx.send(embed=embed)

async def setup(bot):
    await bot.add_cog(CommandName(bot))
